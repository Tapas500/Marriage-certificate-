document.getElementById("certForm").addEventListener("submit", function (e) {

  e.preventDefault();

  const groom = document.getElementById("groom").value;

  const bride = document.getElementById("bride").value;

  const date = document.getElementById("date").value;

  const place = document.getElementById("place").value;

  const regNo = "MRC/" + new Date().getFullYear() + "/" + Math.floor(1000 + Math.random() * 9000);

  const refCode = Math.random().toString(36).substring(2, 10).toUpperCase();

  const refLink = `https://edistrict.tripura.gov.in/F/jcce/${refCode}`;

  document.getElementById("groomOut").textContent = groom;

  document.getElementById("brideOut").textContent = bride;

  document.getElementById("dateOut").textContent = date;

  document.getElementById("placeOut").textContent = place;

  document.getElementById("regNo").textContent = regNo;

  document.getElementById("refNo").textContent = refCode;

  document.getElementById("refLink").textContent = refLink;

  document.getElementById("refLink").href = refLink;

  const qr = new QRious({

    element: document.getElementById("qrCode"),

    value: refLink,

    size: 150

  });

  document.getElementById("output").classList.remove("hidden");

});

function downloadPDF() {

  const element = document.getElementById("certContent");

  html2pdf().from(element).save("Marriage_Certificate.pdf");

}
