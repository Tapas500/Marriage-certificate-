  <script>

    function generateCertificate() {

      const husband = document.getElementById("husband").value;

      const wife = document.getElementById("wife").value;

      const date = document.getElementById("date").value;

      const refNo = `MRC/2025/${Math.floor(10000 + Math.random() * 90000)}`;

      const token = Math.random().toString(36).substring(2, 10).toUpperCase();

      const path = Math.random().toString(36).substring(2, 6);

      const viewLink = `https://edistrict-tripura.netlify.app/t/${path}/${token}`;

      document.getElementById("hname").innerText = husband;

      document.getElementById("wname").innerText = wife;

      document.getElementById("mdate").innerText = date;

      document.getElementById("refBlock").innerHTML = `

        Reference No: ${refNo}<br>

        To View: ${viewLink}<br>

        Token No: ${token}

      `;

      const qr = new QRious({

        element: document.getElementById('qr'),

        value: viewLink,

        size: 100

      });

      document.getElementById("certificate").style.display = "block";

      html2pdf().from(document.getElementById("certificate")).save(`${refNo}.pdf`);

    }

  </script>
